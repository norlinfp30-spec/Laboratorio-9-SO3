Práctica 1: Webmin en Oracle Linux 9.7
Preparación del sistema
sudo dnf update -y
sudo dnf install -y wget curl vim net-tools git tar unzip firewalld
sudo systemctl enable firewalld --now

Verificación del sistema:

cat /etc/oracle-release
uname -r
Instalación de Webmin
sudo tee /etc/yum.repos.d/webmin.repo << 'EOF'
[Webmin]
name=Webmin Distribution Neutral
baseurl=https://download.webmin.com/download/yum
enabled=1
gpgcheck=1
gpgkey=https://download.webmin.com/jcameron-key.asc
EOF
wget -q https://download.webmin.com/jcameron-key.asc
sudo rpm --import jcameron-key.asc
sudo dnf install -y webmin
sudo systemctl enable webmin --now
sudo systemctl status webmin
Configuración del firewall
sudo firewall-cmd --permanent --add-port=10000/tcp
sudo firewall-cmd --reload
sudo firewall-cmd --list-ports
Acceso a Webmin

Abrir en navegador:

https://IP_DE_TU_VM:10000

Credenciales:

Usuario: root
Contraseña: del sistema
Administración de usuarios
id nombreusuario
grep nombreusuario /etc/passwd
Gestión de servicios (SSH)
sudo systemctl start sshd
sudo systemctl stop sshd
sudo systemctl restart sshd
sudo systemctl status sshd
Configuración de red
nmcli device status
ip addr show
sudo nmcli con mod "nombre-conexion" \
ipv4.addresses 192.168.1.100/24 \
ipv4.gateway 192.168.1.1 \
ipv4.dns "8.8.8.8 8.8.4.4" \
ipv4.method manual
sudo nmcli con up "nombre-conexion"
Administración de paquetes
sudo dnf install -y htop
sudo dnf remove -y htop
sudo dnf update -y
Gestión de archivos
ls -la /var/www/
sudo vim /etc/motd
echo "archivo de prueba" | sudo tee /tmp/prueba_webmin.txt
sudo rm /tmp/prueba_webmin.txt
Monitorización del sistema
free -h
top
df -h
vmstat 1 5
Resultado

Webmin instalado y funcional, accesible vía navegador y validado desde terminal.

Práctica 2: Terraform con DigitalOcean
Instalación de Terraform
sudo dnf install -y dnf-plugins-core
sudo tee /etc/yum.repos.d/hashicorp.repo << 'EOF'
[hashicorp]
name=Hashicorp Stable - $basearch
baseurl=https://rpm.releases.hashicorp.com/RHEL/9/$basearch/stable
enabled=1
gpgcheck=1
gpgkey=https://rpm.releases.hashicorp.com/gpg
EOF
sudo dnf install -y terraform
terraform version
Generación de llave SSH
ls -la ~/.ssh/
ssh-keygen -t ed25519 -C "ansible-controller-oracle"
cat ~/.ssh/id_ed25519.pub
Configuración del proyecto
mkdir ~/terraform-do && cd ~/terraform-do
cat << 'EOF' > main.tf
terraform {
  required_providers {
    digitalocean = {
      source  = "digitalocean/digitalocean"
      version = "~> 2.0"
    }
  }
}

provider "digitalocean" {
  token = var.do_token
}

resource "digitalocean_ssh_key" "ansible_key" {
  name       = "ansible-controller-key"
  public_key = var.ssh_public_key
}

resource "digitalocean_droplet" "OS3vm" {
  image    = "ubuntu-22-04-x64"
  name     = "OS3vm"
  region   = "nyc1"
  size     = "s-1vcpu-1gb"
  ssh_keys = [digitalocean_ssh_key.ansible_key.fingerprint]

  tags = ["ansible-client", "practica-os3"]
}

output "vm_ip_address" {
  value = digitalocean_droplet.OS3vm.ipv4_address
}

output "vm_name" {
  value = digitalocean_droplet.OS3vm.name
}
EOF
cat << 'EOF' > variables.tf
variable "do_token" {
  type      = string
  sensitive = true
}

variable "ssh_public_key" {
  type = string
}
EOF
terraform.tfvars (NO subir a GitHub)
LLAVE=$(cat ~/.ssh/id_ed25519.pub)

cat > terraform.tfvars << EOF
do_token       = "TU_TOKEN_AQUI"
ssh_public_key = "$LLAVE"
EOF
terraform.tfvars.example
do_token       = "REEMPLAZAR"
ssh_public_key = "ssh-ed25519 AAAA..."
Ejecución
terraform init
terraform plan
terraform apply
Conexión SSH
ssh -i ~/.ssh/id_ed25519 root@IP_GENERADA
Verificación
hostname
uname -a
cat /etc/os-release
free -h
nproc
df -h
ip addr show
Estado de infraestructura
terraform state list
terraform state show digitalocean_droplet.OS3vm
terraform output
Eliminación de recursos
terraform destroy
Práctica 3: Instalación y Configuración de Ansible
1. Preparar el Ansible Controller (Oracle Linux)
Ejecuta esto para asegurar que tienes todas las herramientas:

Bash
sudo dnf install epel-release -y
sudo dnf install ansible python3-pip -y
sudo pip3 install pywinrm
2. Configurar el Inventario
Copia y pega este bloque para configurar las IPs correctas:

Bash
sudo tee /etc/ansible/hosts << 'EOF'
[linux]
ansible1 ansible_host=67.205.145.199 ansible_user=ansible

[win]
ansible2 ansible_host=192.168.0.195 ansible_user=ansible ansible_password=Ansible123! ansible_connection=winrm ansible_winrm_server_cert_validation=ignore
EOF
3. Crear usuario en Ubuntu (DigitalOcean)
Entra a tu Droplet para crear el usuario que usará Ansible:

Bash
ssh -i ~/.ssh/id_ed25519 root@67.205.145.199

# Dentro de Ubuntu:
adduser ansible --disabled-password --gecos ""
echo "ansible:Ansible123!" | chpasswd
usermod -aG sudo ansible
echo "ansible ALL=(ALL) NOPASSWD:ALL" | tee /etc/sudoers.d/ansible
mkdir -p /home/ansible/.ssh
cp /root/.ssh/authorized_keys /home/ansible/.ssh/
chown -R ansible:ansible /home/ansible/.ssh
exit
4. Pruebas de conexión (Ping)
IMPORTANTE: Hazlas por separado para evitar errores de protocolo.

Bash
# Probar Linux
ansible linux -m ping

# Probar Windows
ansible win -m win_ping
Práctica 4: Comandos Ad-Hoc

# Crear un archivo y moverlo

Bash
ansible ansible1 -m reboot --become
Práctica 5: Playbooks
Crea estos dos archivos en tu carpeta actual en Oracle Linux:

1. Playbook para NGINX (nginx.yml)
Bash
cat << 'EOF' > nginx.yml
---
- name: Instalacion de NGINX en Ubuntu
  hosts: linux
  become: yes
  tasks:
    - name: Instalar nginx
      apt:
        name: nginx
        state: present
        update_cache: yes
    - name: Iniciar servicio
      service:
        name: nginx
        state: started
        enabled: yes
EOF
Ejecutar: ansible-playbook nginx.yml

2. Playbook para Notepad++ (notepad.yml)
Bash
cat << 'EOF' > notepad.yml
---
- name: Instalar Notepad++ en Windows
  hosts: win
  tasks:
    - name: Descargar e instalar Notepad++
      win_package:
        path: https://github.com/notepad-plus-plus/notepad-plus-plus/releases/download/v8.6.5/npp.8.6.5.Installer.x64.exe
        product_id: 'Notepad++'
        arguments: /S
        state: present
EOF
Ejecutar: ansible-playbook notepad.yml
